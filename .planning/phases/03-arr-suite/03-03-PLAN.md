```bash
cmd_blocklist() {
    local app="${1:-}"
    [[ -z "$app" ]] && die "Usage: arr blocklist <sonarr|radarr|lidarr>"
    app=$(resolve_app "$app")

    case "$app" in
        sonarr|radarr|lidarr) ;;
        *) die "Blocklist not supported for $app (only sonarr, radarr, lidarr)" ;;
    esac

    local response
    response=$(arr_api "$app" GET "/blocklist?page=1&pageSize=50&sortKey=date&sortDirection=descending") || die "Can't connect to $app"

    echo -e "  ${BOLD}${app}${RESET} — blocklist:"
    echo "$response" | python3 -c "
import sys, json
data = json.load(sys.stdin)
records = data.get('records', [])
total = data.get('totalRecords', 0)
if not records:
    print('    (empty)')
    sys.exit()
print(f'    {total} total items (showing up to 50)')
print()
print(f'    {\"ID\":<8} {\"Title\":<45} {\"Date\":<12} {\"Reason\"}')
print(f'    {\"--\":<8} {\"-----\":<45} {\"----\":<12} {\"------\"}')
for r in records:
    bid = r.get('id', '?')
    title = r.get('sourceTitle', '?')[:44]
    date = r.get('date', '?')[:10]
    # Get indexer or message for reason
    msg_list = r.get('message', '')
    if not msg_list:
        msgs = r.get('statusMessages', [])
        msg_list = msgs[0].get('messages', [''])[0][:30] if msgs else ''
    else:
        msg_list = msg_list[:30]
    print(f'    {bid:<8} {title:<45} {date:<12} {msg_list}')
" 2>/dev/null
}
```

**2. Add cmd_blocklist_clear() — implements ARR-11.**

```bash
cmd_blocklist_clear() {
    local app="${1:-}"
    [[ -z "$app" ]] && die "Usage: arr blocklist-clear <sonarr|radarr|lidarr>"
    app=$(resolve_app "$app")

    case "$app" in
        sonarr|radarr|lidarr) ;;
        *) die "Blocklist not supported for $app (only sonarr, radarr, lidarr)" ;;
    esac

    # Fetch all blocklist IDs
    local response
    response=$(arr_api "$app" GET "/blocklist?page=1&pageSize=1000") || die "Can't connect to $app"

    local ids
    ids=$(echo "$response" | python3 -c "
import sys, json
data = json.load(sys.stdin)
records = data.get('records', [])
ids = [r['id'] for r in records if 'id' in r]
if not ids:
    print('')
else:
    print(','.join(str(i) for i in ids))
" 2>/dev/null)

    if [[ -z "$ids" ]]; then
        ok "  $app: blocklist is empty, nothing to clear."
        return
    fi

    local count
    count=$(echo "$ids" | tr ',' '\n' | wc -l)

    confirm_action "Clear $count blocklist items from $app?"

    local payload
    payload=$(IDS="$ids" python3 -c "
import os, json
ids = [int(i) for i in os.environ['IDS'].split(',')]
print(json.dumps({'ids': ids}))
")

    info "Clearing blocklist on $app..."
    arr_api "$app" DELETE "/blocklist/bulk" "$payload" || die "Failed to clear blocklist on $app"
    ok "  Cleared $count items from $app blocklist."
}
```

**2. Add cmd_logs() — implements ARR-13.**

```bash
cmd_logs() {
    local app="${1:-}" count="${2:-25}"
    [[ -z "$app" ]] && die "Usage: arr logs <sonarr|radarr|lidarr|prowlarr> [count]"
    app=$(resolve_app "$app")

    case "$app" in
        sonarr|radarr|lidarr|prowlarr) ;;
        *) die "Logs not supported for $app" ;;
    esac

    local response
    response=$(arr_api "$app" GET "/log?page=1&pageSize=${count}&sortKey=time&sortDirection=descending") || die "Can't connect to $app"

    echo -e "  ${BOLD}${app}${RESET} — recent logs (${count}):"
    echo "$response" | python3 -c "
import sys, json
data = json.load(sys.stdin)
records = data.get('records', [])
if not records:
    print('    (no logs)')
    sys.exit()
for r in records:
    time = r.get('time', '?')[:19].replace('T', ' ')
    level = r.get('level', '?')
    logger = r.get('logger', '?').split('.')[-1][:20]
    msg = r.get('message', '?')[:80]
    if level == 'error':
        color = '\033[31m'
    elif level == 'warn':
        color = '\033[33m'
    else:
        color = ''
    reset = '\033[0m' if color else ''
    print(f'    {color}{time}  {level:<6} {logger:<20} {msg}{reset}')
" 2>/dev/null
}
```

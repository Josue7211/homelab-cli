```bash
cmd_profiles() {
    local app="${1:-}"
    [[ -z "$app" ]] && die "Usage: arr profiles <sonarr|radarr|lidarr>"
    app=$(resolve_app "$app")

    case "$app" in
        sonarr|radarr|lidarr) ;;
        *) die "Profiles not supported for $app (only sonarr, radarr, lidarr)" ;;
    esac

    local response
    response=$(arr_api "$app" GET "/qualityprofile") || die "Can't connect to $app"

    echo -e "  ${BOLD}${app}${RESET} — quality profiles:"
    echo "$response" | python3 -c "
import sys, json
data = json.load(sys.stdin)
if not data:
    print('    (none)')
    sys.exit()
print(f'    {\"ID\":<6} {\"Name\":<30} {\"Cutoff\":<20} {\"Items\"}')
print(f'    {\"--\":<6} {\"----\":<30} {\"------\":<20} {\"-----\"}')
for p in data:
    pid = p.get('id', '?')
    name = p.get('name', '?')
    cutoff_id = p.get('cutoff', 0)
    items = p.get('items', [])
    # Find cutoff name from items
    cutoff_name = str(cutoff_id)
    for item in items:
        if item.get('quality', {}).get('id') == cutoff_id:
            cutoff_name = item['quality'].get('name', str(cutoff_id))
            break
        for sub in item.get('items', []):
            if sub.get('quality', {}).get('id') == cutoff_id:
                cutoff_name = sub['quality'].get('name', str(cutoff_id))
                break
    count = len([i for i in items if i.get('allowed')])
    print(f'    {pid:<6} {name:<30} {cutoff_name:<20} {count} allowed')
" 2>/dev/null
}
```

```bash
cmd_tag_add() {
    local app="${1:-}" label="${2:-}"
    [[ -z "$app" || -z "$label" ]] && die "Usage: arr tag-add <sonarr|radarr|lidarr|prowlarr> <label>"
    app=$(resolve_app "$app")

    case "$app" in
        sonarr|radarr|lidarr|prowlarr) ;;
        *) die "Tags not supported for $app" ;;
    esac

    local payload
    payload=$(LABEL="$label" python3 -c "
import json, os
print(json.dumps({'label': os.environ['LABEL']}))
")

    info "Creating tag '$label' on $app..."
    local response
    response=$(arr_api "$app" POST "/tag" "$payload") || die "Failed to create tag on $app"
    echo "$response" | python3 -c "
import sys, json
d = json.load(sys.stdin)
print(f\"    Created tag: id={d.get('id', '?')} label={d.get('label', '?')}\")
" 2>/dev/null
    ok "  Done."
}
```

```bash
cmd_indexers() {
    local app="${1:-}"
    local apps_to_check=()

    if [[ -n "$app" ]]; then
        apps_to_check=("$(resolve_app "$app")")
    else
        apps_to_check=("${ALL_APPS[@]}")
    fi

    for app in "${apps_to_check[@]}"; do
        local response
        response=$(arr_api "$app" GET "/indexer" 2>/dev/null) || {
            warn "  $app: could not connect"
            continue
        }

        echo -e "  ${BOLD}${app}${RESET} — indexers:"
        echo "$response" | python3 -c "
import sys, json
data = json.load(sys.stdin)
if not data:
    print('    (none)')
    sys.exit()
print(f'    {\"ID\":<6} {\"Name\":<30} {\"Protocol\":<12} {\"Enabled\"}')
print(f'    {\"--\":<6} {\"----\":<30} {\"--------\":<12} {\"-------\"}')
for ix in data:
    iid = ix.get('id', '?')
    name = ix.get('name', '?')
    proto = ix.get('protocol', '?')
    enabled = 'yes' if ix.get('enableRss', ix.get('enable', False)) else 'no'
    print(f'    {iid:<6} {name:<30} {proto:<12} {enabled}')
" 2>/dev/null
        echo ""
    done
}
```

**resolve_item_id()** helper — resolves name-or-ID to a bw item ID:
- If argument looks like a UUID (matches pattern with hyphens and hex chars), use it directly
- Otherwise, run `bw get item "$name" 2>/dev/null` and extract the `id` field with python3
- Die with "Item '$name' not found" on failure
- Print the resolved ID to stdout

```bash
resolve_item_id() {
    local input="$1"
    # If it looks like a UUID, use directly
    if [[ "$input" =~ ^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$ ]]; then
        echo "$input"
        return
    fi
    # Resolve name to ID
    local item_json
    item_json=$(bw get item "$input" 2>/dev/null) || die "Item '$input' not found"
    echo "$item_json" | python3 -c "import sys,json; print(json.load(sys.stdin)['id'])"
}
```

**cmd_create()** (per D-01, BW-01):
- Usage: `vault create <name> <username> <password> [--url URL] [--folder FOLDER]`
- Parse positional args: name=$1, username=$2, password=$3
- Parse optional flags: loop through remaining args for --url and --folder
- Die if fewer than 3 positional args: `die "Usage: vault create <name> <username> <password> [--url URL] [--folder FOLDER]"`
- Step 1: `check_unlocked`
- Step 2: Get item template: `bw get template item 2>/dev/null`
- Step 3: Use python3 with os.environ to safely build the JSON (set NAME, USERNAME, PASSWORD, URI, FOLDER_ID as env vars, then python3 reads them to avoid shell injection):
  - Set item type to 1 (login)
  - Set name from env
  - Build login object with username, password from env
  - If URI provided, add uris array with one entry (match=0, uri=value)
  - If FOLDER provided, resolve folder name to ID: `bw list folders 2>/dev/null` then match by name, set folderId
- Step 4: Pipe the modified JSON through `bw encode` then `bw create item`
- Step 5: Extract and print new item ID from the bw create response
- Print: `ok "Created item: $name (ID: $new_id)"`

```bash
cmd_create() {
    local name="" username="" password="" url="" folder=""

    # Parse positional args
    [[ $# -lt 3 ]] && die "Usage: vault create <name> <username> <password> [--url URL] [--folder FOLDER]"
    name="$1"; username="$2"; password="$3"
    shift 3

    # Parse optional flags
    while [[ $# -gt 0 ]]; do
        case "$1" in
            --url)    url="${2:-}"; shift 2 ;;
            --folder) folder="${2:-}"; shift 2 ;;
            *)        die "Unknown option: $1" ;;
        esac
    done

    check_unlocked

    # Resolve folder name to ID if specified
    local folder_id=""
    if [[ -n "$folder" ]]; then
        folder_id=$(FOLDER="$folder" bw list folders 2>/dev/null | python3 -c "
import sys, json, os
folders = json.load(sys.stdin)
target = os.environ['FOLDER'].lower()
for f in folders:
    if f.get('name', '').lower() == target:
        print(f['id'])
        sys.exit()
print('')
") || die "Failed to list folders"
        [[ -z "$folder_id" ]] && die "Folder '$folder' not found"
    fi

    # Get template, modify, encode, create
    local template
    template=$(bw get template item 2>/dev/null) || die "Failed to get item template"

    local encoded
    encoded=$(NAME="$name" USERNAME="$username" PASSWORD="$password" URI="$url" FOLDER_ID="$folder_id" \
        python3 -c "
import sys, json, os
item = json.loads(sys.stdin.read())
item['type'] = 1  # login
item['name'] = os.environ['NAME']
item['login'] = {
    'username': os.environ['USERNAME'],
    'password': os.environ['PASSWORD'],
    'uris': []
}
uri = os.environ.get('URI', '')
if uri:
    item['login']['uris'] = [{'match': 0, 'uri': uri}]
fid = os.environ.get('FOLDER_ID', '')
if fid:
    item['folderId'] = fid
else:
    item['folderId'] = None
import base64
print(base64.b64encode(json.dumps(item).encode()).decode())
" <<< "$template") || die "Failed to build item JSON"

    local result
    result=$(echo "$encoded" | bw create item 2>/dev/null) || die "Failed to create item"

    local new_id
    new_id=$(echo "$result" | python3 -c "import sys,json; print(json.load(sys.stdin)['id'])" 2>/dev/null || echo "?")

    ok "  Created item: $name (ID: $new_id)"
}
```

**cmd_edit()** (per D-02, BW-02):
- Usage: `vault edit <id|name> [--username U] [--password P] [--url URL] [--notes N]`
- Parse first arg as item identifier (required)
- Parse optional flags for each field that can be changed
- Die if no identifier: `die "Usage: vault edit <id|name> [--username U] [--password P] [--url URL] [--notes N]"`
- Die if no field flags provided: `die "No fields to edit. Use --username, --password, --url, or --notes"`
- Step 1: `check_unlocked`
- Step 2: `resolve_item_id` to get the item ID
- Step 3: `bw get item "$item_id" 2>/dev/null` to get current item JSON
- Step 4: Use python3 with os.environ to patch only the specified fields (set EDIT_USERNAME, EDIT_PASSWORD, EDIT_URL, EDIT_NOTES env vars -- empty string means "not specified", use a sentinel "_UNSET_" for truly not provided):
  - If username provided: update login.username
  - If password provided: update login.password
  - If url provided: replace or add uris[0]
  - If notes provided: update notes
- Step 5: Pipe through `bw encode` then `bw edit item "$item_id"`
- Print: `ok "Updated item: $item_id"`

```bash
cmd_edit() {
    [[ $# -lt 1 ]] && die "Usage: vault edit <id|name> [--username U] [--password P] [--url URL] [--notes N]"

    local identifier="$1"; shift
    local new_username="_UNSET_" new_password="_UNSET_" new_url="_UNSET_" new_notes="_UNSET_"

    while [[ $# -gt 0 ]]; do
        case "$1" in
            --username) new_username="${2:-}"; shift 2 ;;
            --password) new_password="${2:-}"; shift 2 ;;
            --url)      new_url="${2:-}"; shift 2 ;;
            --notes)    new_notes="${2:-}"; shift 2 ;;
            *)          die "Unknown option: $1" ;;
        esac
    done

    # Check at least one field specified
    if [[ "$new_username" == "_UNSET_" && "$new_password" == "_UNSET_" && "$new_url" == "_UNSET_" && "$new_notes" == "_UNSET_" ]]; then
        die "No fields to edit. Use --username, --password, --url, or --notes"
    fi

    check_unlocked

    local item_id
    item_id=$(resolve_item_id "$identifier")

    local item_json
    item_json=$(bw get item "$item_id" 2>/dev/null) || die "Failed to get item '$identifier'"

    local encoded
    encoded=$(EDIT_USERNAME="$new_username" EDIT_PASSWORD="$new_password" EDIT_URL="$new_url" EDIT_NOTES="$new_notes" \
        python3 -c "
import sys, json, os, base64
item = json.load(sys.stdin)
login = item.get('login', {}) or {}

u = os.environ.get('EDIT_USERNAME', '_UNSET_')
if u != '_UNSET_':
    login['username'] = u

p = os.environ.get('EDIT_PASSWORD', '_UNSET_')
if p != '_UNSET_':
    login['password'] = p

url = os.environ.get('EDIT_URL', '_UNSET_')
if url != '_UNSET_':
    login['uris'] = [{'match': 0, 'uri': url}]

item['login'] = login

n = os.environ.get('EDIT_NOTES', '_UNSET_')
if n != '_UNSET_':
    item['notes'] = n

print(base64.b64encode(json.dumps(item).encode()).decode())
" <<< "$item_json") || die "Failed to modify item JSON"

    echo "$encoded" | bw edit item "$item_id" >/dev/null 2>&1 || die "Failed to edit item '$identifier'"

    ok "  Updated item: $identifier"
}
```

```bash
cmd_delete() {
    local identifier="${1:-}"
    [[ -z "$identifier" ]] && die "Usage: vault delete <id|name>"

    check_unlocked

    local item_id
    item_id=$(resolve_item_id "$identifier")

    # Get item name for confirmation message
    local item_name
    item_name=$(bw get item "$item_id" 2>/dev/null | python3 -c "import sys,json; print(json.load(sys.stdin).get('name','?'))" 2>/dev/null || echo "$identifier")

    confirm_action "Delete item '$item_name'?"

    bw delete item "$item_id" 2>/dev/null || die "Failed to delete item '$identifier'"

    ok "  Deleted item: $item_name ($item_id)"
}
```

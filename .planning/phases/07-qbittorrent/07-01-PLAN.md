```bash
qbt_get() {
    local path="$1"
    qbt_login
    curl -sf -b "$_cookie_file" "${QBIT_URL}${path}" 2>/dev/null
}

qbt_post() {
    local path="$1" data="${2:-}"
    qbt_login
    if [[ -n "$data" ]]; then
        curl -sf -b "$_cookie_file" -X POST "${QBIT_URL}${path}" -d "$data" 2>/dev/null
    else
        curl -sf -b "$_cookie_file" -X POST "${QBIT_URL}${path}" 2>/dev/null
    fi
}
```

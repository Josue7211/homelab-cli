```bash
overseerr_api() {
    local method="$1" path="$2" data="${3:-}"
    local url="${OVERSEERR_URL}/api/v1${path}"
    local key
    key=$(get_overseerr_key)

    case "$method" in
        GET)    api_get "$url" "$key" ;;
        POST)   api_post "$url" "$key" "$data" ;;
        DELETE) api_delete "$url" "$key" ;;
    esac
}
```

- Fetch both Radarr and Sonarr service configs:
  - GET `/service/radarr` for Radarr instances
  - GET `/service/sonarr` for Sonarr instances

  - For each service type, show a sub-header ("Radarr:", "Sonarr:")

  - If a service type returns empty or fails, show "(none configured)"

```bash
cmd_services() {
    header "CONNECTED SERVICES"

    # Fetch Radarr services
    local radarr
    radarr=$(overseerr_api GET "/service/radarr") 2>/dev/null

    echo "  Radarr:"
    if [[ -n "$radarr" ]]; then
        echo "$radarr" | python3 -c "
import sys, json
data = json.load(sys.stdin)
if not data:
    print('    (none configured)')
    sys.exit()
for s in data:
    sid = s.get('id', '?')
    name = s.get('name', '?')
    host = s.get('hostname', s.get('baseUrl', '?'))
    port = s.get('port', '')
    url = f'{host}:{port}' if port else host
    active = s.get('isActive', False)
    default = s.get('isDefault', False)
    profile = s.get('activeProfileName', s.get('activeProfileId', '-'))
    active_str = '\033[32mactive\033[0m' if active else '\033[2minactive\033[0m'
    default_str = ' (default)' if default else ''
    print(f'    #{sid} {name:<20} {url:<35} {active_str}{default_str}  profile: {profile}')
" 2>/dev/null || echo "    (none configured)"
    else
        echo "    (none configured)"
    fi

    echo ""

    # Fetch Sonarr services
    local sonarr
    sonarr=$(overseerr_api GET "/service/sonarr") 2>/dev/null

    echo "  Sonarr:"
    if [[ -n "$sonarr" ]]; then
        echo "$sonarr" | python3 -c "
import sys, json

    print('    (none configured)')
```

**cmd_notifications()** (per D-05, OVR-05):
- Usage: `overseerr notifications` (no args)
- GET `/settings/notifications/discord` — Discord agent
- GET `/settings/notifications/email` — Email agent
- GET `/settings/notifications/slack` — Slack agent
- GET `/settings/notifications/telegram` — Telegram agent
- GET `/settings/notifications/pushover` — Pushover agent
- GET `/settings/notifications/webhook` — Webhook agent
- Display with header "NOTIFICATION AGENTS":
  - For each agent: name, enabled status (green "enabled" / dim "disabled"), and a brief config summary
  - Discord: show webhookUrl presence (masked)
  - Email: show sender address
  - Slack: show webhook presence
  - Telegram: show bot configured
  - Pushover: show configured
  - Webhook: show URL presence
- If an agent fetch fails, show it as "unavailable"

```bash
cmd_notifications() {
    header "NOTIFICATION AGENTS"

    local agents="discord email slack telegram pushover webhook"
    local agent_labels="Discord Email Slack Telegram Pushover Webhook"

    local i=0
    for agent in $agents; do
        i=$((i + 1))
        local label
        label=$(echo "$agent_labels" | cut -d' ' -f$i)

        local data
        data=$(overseerr_api GET "/settings/notifications/${agent}") 2>/dev/null

        if [[ -n "$data" ]]; then
            echo "$data" | AGENT="$label" python3 -c "
import sys, json, os
d = json.load(sys.stdin)
agent = os.environ['AGENT']
enabled = d.get('enabled', False)
status = '\033[32menabled\033[0m' if enabled else '\033[2mdisabled\033[0m'

# Config summary varies by agent type
summary = ''
agent_lower = agent.lower()
if agent_lower == 'discord':
    wh = d.get('options', {}).get('webhookUrl', '')
    summary = 'webhook configured' if wh else 'no webhook'
elif agent_lower == 'email':
    sender = d.get('options', {}).get('senderName', d.get('options', {}).get('emailFrom', ''))
    summary = f'from: {sender}' if sender else 'not configured'
elif agent_lower == 'slack':
    wh = d.get('options', {}).get('webhookUrl', '')
    summary = 'webhook configured' if wh else 'no webhook'
elif agent_lower == 'telegram':
    token = d.get('options', {}).get('botAPI', d.get('options', {}).get('botToken', ''))
    summary = 'bot configured' if token else 'no bot token'
elif agent_lower == 'pushover':
    key = d.get('options', {}).get('accessToken', d.get('options', {}).get('appToken', ''))
    summary = 'configured' if key else 'not configured'
elif agent_lower == 'webhook':
    url = d.get('options', {}).get('webhookUrl', d.get('options', {}).get('jsonPayload', ''))
    summary = 'URL configured' if url else 'not configured'

print(f'  {agent:<12} {status:<25} {summary}')
" 2>/dev/null || echo "  ${label}$(printf '%*s' $((12 - ${#label})) '') unavailable"
        else
            printf "  %-12s \033[2munavailable\033[0m\n" "$label"
        fi
    done
}
```

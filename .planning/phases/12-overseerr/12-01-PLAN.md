**cmd_approve_all()** (per D-01, OVR-01):
- Usage: `overseerr approve-all` (no args)
- Step 1: GET `/request?filter=pending&take=100` to fetch all pending requests
- Step 2: Parse response with python3 to extract count and list of request IDs
- Step 3: If no pending requests, print "No pending requests" and return
- Step 4: Display pending count and call `confirm_action "Approve all $count pending requests?"` (required -- bulk operation)
- Step 5: Loop through each request ID and POST `/request/{id}/approve`
- Step 6: Show count of approved vs failed
- Die if initial fetch fails: `die "Cannot connect to Overseerr"`

```bash
cmd_approve_all() {
    header "BULK APPROVE REQUESTS"

    local response
    response=$(overseerr_api GET "/request?filter=pending&take=100") || die "Cannot connect to Overseerr"

    # Extract pending request IDs
    local ids count
    ids=$(echo "$response" | python3 -c "
import sys, json
data = json.load(sys.stdin)
results = data.get('results', [])
for r in results:
    print(r.get('id', ''))
") || die "Failed to parse requests"

    count=$(echo "$ids" | grep -c '[0-9]' || true)

    if [[ "$count" -eq 0 ]]; then
        info "  No pending requests to approve."
        return
    fi

    # Show what will be approved
    echo "$response" | python3 -c "
import sys, json
data = json.load(sys.stdin)
results = data.get('results', [])
for r in results:
    rid = r.get('id', '?')
    media = r.get('media', {})
    title = media.get('title', media.get('originalTitle', ''))
    if not title:
        title = f'Request #{rid}'
    rtype = r.get('type', '?')
    user = r.get('requestedBy', {}).get('displayName', r.get('requestedBy', {}).get('email', '?'))
    print(f'  #{rid:<5} {rtype:<6} {title:<40} by {user}')
"
    echo ""

    confirm_action "Approve all $count pending requests?"

    local approved=0 failed=0
    while IFS= read -r id; do
        [[ -z "$id" ]] && continue
        if overseerr_api POST "/request/${id}/approve" >/dev/null 2>&1; then
            ((approved++))
        else
            warn "  Failed to approve request #$id"
            ((failed++))
        fi
    done <<< "$ids"

    ok "  Approved: $approved"
    [[ "$failed" -gt 0 ]] && warn "  Failed: $failed"
}
```

**cmd_users()** (per D-02, OVR-02):
- Usage: `overseerr users` (no args)
- GET `/user?take=50` to fetch all users

  - ID, Display Name, Email, User Type (plex/local/jellyfin), Request Count, Role (admin/user)
- Map userType: 1=Plex, 2=Local, 3=Jellyfin
- Map permissions: if permissions & 2 (ADMIN bit) -> "admin", else "user"
- Show total count at top
- Die if fetch fails: `die "Cannot connect to Overseerr"`

```bash
cmd_users() {
    header "OVERSEERR USERS"

    local response
    response=$(overseerr_api GET "/user?take=50") || die "Cannot connect to Overseerr"

    echo "$response" | python3 -c "
import sys, json
data = json.load(sys.stdin)
results = data.get('results', [])
total = data.get('pageInfo', {}).get('results', len(results))
if not results:
    print('  No users found.')
    sys.exit()
print(f'  Total: {total} users\n')
print(f'  {\"ID\":<6} {\"Name\":<20} {\"Email\":<30} {\"Type\":<10} {\"Requests\":<10} {\"Role\":<8}')
print(f'  {\"─\"*6} {\"─\"*20} {\"─\"*30} {\"─\"*10} {\"─\"*10} {\"─\"*8}')
type_map = {1: 'Plex', 2: 'Local', 3: 'Jellyfin'}
for u in results:
    uid = u.get('id', '?')
    name = (u.get('displayName', '') or u.get('username', '') or '?')[:20]
    email = (u.get('email', '') or '-')[:30]
    utype = type_map.get(u.get('userType', 0), '?')
    reqs = u.get('requestCount', 0)
    perms = u.get('permissions', 0)
    role = 'admin' if perms & 2 else 'user'
    # Color admin role
    role_str = f'\033[33m{role}\033[0m' if role == 'admin' else role
    print(f'  {uid:<6} {name:<20} {email:<30} {utype:<10} {reqs:<10} {role_str}')
"
}
```

**cmd_request_detail()** (per D-03, OVR-03):
- Usage: `overseerr request-detail <id>` (named request-detail to avoid conflict with existing `request` command that creates requests)
- Validate arg present or die: `die "Usage: overseerr request-detail <request_id>"`
- GET `/request/${id}` to fetch full request detail
- Display with header "REQUEST #$id":
  - Request ID, Type (movie/tv), Status (PENDING/APPROVED/DECLINED with color)
  - Media title (from media.title or media.originalTitle, or "Unknown" with tmdbId/tvdbId fallback)
  - Media status (unknown/pending/processing/partially available/available)
  - Requester (displayName or email from requestedBy object)
  - Created date, Modified date (truncate to date only, first 10 chars)
  - For TV: show requested seasons if present in the `seasons` array (season numbers)
- Die if fetch fails: `die "Failed to fetch request #$id"`

```bash
cmd_request_detail() {
    local id="${1:-}"
    [[ -z "$id" ]] && die "Usage: overseerr request-detail <request_id>"

    local response
    response=$(overseerr_api GET "/request/${id}") || die "Failed to fetch request #$id"

    header "REQUEST #$id"

    echo "$response" | python3 -c "
import sys, json
d = json.load(sys.stdin)

rid = d.get('id', '?')
rtype = d.get('type', '?')

# Status
status_val = d.get('status', 0)
status_map = {1: 'PENDING', 2: 'APPROVED', 3: 'DECLINED'}
status_str = status_map.get(status_val, str(status_val))
status_colors = {'PENDING': '\033[33m', 'APPROVED': '\033[32m', 'DECLINED': '\033[31m'}
color = status_colors.get(status_str, '')
print(f'  Status:      {color}{status_str}\033[0m')
print(f'  Type:        {rtype}')

# Media info
media = d.get('media', {})
title = media.get('title', media.get('originalTitle', ''))
if not title:
    tmdb = media.get('tmdbId', '')
    tvdb = media.get('tvdbId', '')
    title = f'tmdb:{tmdb}' if tmdb else (f'tvdb:{tvdb}' if tvdb else 'Unknown')
print(f'  Title:       {title}')

# Media status
media_status_val = media.get('status', 0)
media_status_map = {1: 'unknown', 2: 'pending', 3: 'processing', 4: 'partially available', 5: 'available'}
media_status = media_status_map.get(media_status_val, str(media_status_val))
ms_color = '\033[32m' if media_status == 'available' else ('\033[33m' if media_status in ('pending', 'processing') else '')
print(f'  Media:       {ms_color}{media_status}\033[0m')

# Requester
requester = d.get('requestedBy', {})
req_name = requester.get('displayName', requester.get('email', '?'))
print(f'  Requested by: {req_name}')

# Dates
created = d.get('createdAt', '?')[:10]
modified = d.get('updatedAt', d.get('modifiedAt', '?'))[:10] if d.get('updatedAt', d.get('modifiedAt')) else '-'
print(f'  Created:     {created}')
print(f'  Modified:    {modified}')

# Seasons (TV only)
seasons = d.get('seasons', [])
if seasons:
    season_nums = sorted([s.get('seasonNumber', '?') for s in seasons])
    print(f'  Seasons:     {', '.join(str(s) for s in season_nums)}')
"
}
```

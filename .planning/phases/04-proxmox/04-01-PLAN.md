1. Replace the existing `pve_api()` function (around line 28-34) with a multi-method version per D-01 and D-03:

```bash
pve_api() {
    local method="${1:-GET}"
    local path="$2"
    shift 2
    local token
    token=$(get_pve_token)
    local -a curl_args=(
        -sk
        -X "$method"
        "${PVE_HOST}/api2/json${path}"
        -H "Authorization: PVEAPIToken=${PVE_TOKEN}=${token}"
    )
    # Remaining args are passed through (e.g., -d key=val for form data)
    curl_args+=("$@")
    curl "${curl_args[@]}"
}
```

Key design: default method is GET. Extra arguments after path are passed directly to curl, enabling `-d "key=val"` for form-encoded POST/PUT bodies. No Content-Type header needed -- Proxmox accepts form-encoded by default.

cmd_config() {
    local vmid="$1"
    local vm_type
    vm_type=$(_resolve_vm_type "$vmid")
    header "VM $vmid Configuration"
    pve_api GET "/nodes/${PVE_NODE}/$vm_type/$vmid/config" | python3 -c "
import sys, json
data = json.load(sys.stdin)['data']

# Group keys by category
cpu_keys = ['cores', 'sockets', 'cpu', 'vcpus', 'cpulimit', 'cpuunits', 'numa', 'affinity']
mem_keys = ['memory', 'balloon', 'shares']
disk_prefixes = ('scsi', 'virtio', 'ide', 'sata', 'efidisk', 'tpmstate')
net_prefixes = ('net',)
groups = {'CPU': {}, 'MEMORY': {}, 'DISK': {}, 'NETWORK': {}, 'OTHER': {}}
for k, v in sorted(data.items()):
    if k == 'digest': continue
    if k in cpu_keys: groups['CPU'][k] = v
    elif k in mem_keys: groups['MEMORY'][k] = v
    elif any(k.startswith(p) for p in disk_prefixes): groups['DISK'][k] = v
    elif any(k.startswith(p) for p in net_prefixes): groups['NETWORK'][k] = v
    else: groups['OTHER'][k] = v

for group, items in groups.items():
    if not items: continue
    print(f'\n  {group}:')
    for k, v in sorted(items.items()):
        print(f'    {k:<20} {v}')
"
}

2. **cmd_set** (per D-05, PVE-02): Modify VM config with form-encoded key-value pairs. No confirmation needed (reversible).

```bash
cmd_set() {
    local vmid="$1"; shift
    [[ $# -eq 0 ]] && die "Usage: homelab set <vmid> KEY=VAL..."
    local vm_type
    vm_type=$(_resolve_vm_type "$vmid")
    local -a data_args=()
    for kv in "$@"; do
        [[ "$kv" == *=* ]] || die "Invalid argument '$kv' — expected KEY=VAL"
        data_args+=(-d "$kv")
    done
    info "Updating VM $vmid config..."
    pve_api PUT "/nodes/${PVE_NODE}/$vm_type/$vmid/config" "${data_args[@]}" > /dev/null
    ok "Done."
}
```

3. **cmd_resize** (per D-06, PVE-03): Resize VM disk.

cmd_resize() {
    local vmid="$1" disk="$2" size="$3"

    info "Resizing $disk on VM $vmid by $size..."

4. **cmd_clone** (per D-07 and D-08, PVE-04): Clone VM with optional --name and --target.

```bash
cmd_clone() {
    local vmid="$1"; shift
    local vm_type name="" target=""
    vm_type=$(_resolve_vm_type "$vmid")
    while [[ $# -gt 0 ]]; do
        case "$1" in
            --name)   name="$2"; shift 2 ;;
            --target) target="$2"; shift 2 ;;
            *) die "Unknown option: $1" ;;
        esac
    done
    local -a data_args=(-d "full=1")
    [[ -n "$name" ]] && data_args+=(-d "name=$name")
    [[ -n "$target" ]] && data_args+=(-d "target=$target")
    info "Cloning VM $vmid (full clone)..."
    local result
    result=$(pve_api POST "/nodes/${PVE_NODE}/$vm_type/$vmid/clone" "${data_args[@]}")
    local upid
    upid=$(echo "$result" | python3 -c "import sys,json; print(json.load(sys.stdin).get('data',''))" 2>/dev/null || echo "")
    if [[ -n "$upid" ]]; then
        ok "Clone started. Task: $upid"
    else
        ok "Clone started."
    fi
}
```

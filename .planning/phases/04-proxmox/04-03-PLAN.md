1. **cmd_backup** (per D-14, PVE-10): Trigger vzdump backup with UPID reporting.

```bash
cmd_backup() {
    local vmid="$1"; shift
    local vm_type storage="" compress="zstd"
    vm_type=$(_resolve_vm_type "$vmid")
    while [[ $# -gt 0 ]]; do
        case "$1" in
            --storage)  storage="$2"; shift 2 ;;
            --compress) compress="$2"; shift 2 ;;
            *) die "Unknown option: $1" ;;
        esac
    done
    local -a data_args=(-d "vmid=$vmid" -d "compress=$compress" -d "mode=snapshot")
    [[ -n "$storage" ]] && data_args+=(-d "storage=$storage")
    info "Starting backup for VM $vmid (compress=$compress)..."
    local result
    result=$(pve_api POST "/nodes/${PVE_NODE}/vzdump" "${data_args[@]}")
    local upid
    upid=$(echo "$result" | python3 -c "import sys,json; print(json.load(sys.stdin).get('data',''))" 2>/dev/null || echo "")
    if [[ -n "$upid" ]]; then
        ok "Backup started. Task: $upid"
        info "Track with: homelab tasks"
    else
        ok "Backup started."
    fi
}
```

Note: The vzdump endpoint is `/nodes/{node}/vzdump` (not nested under VM type), but requires `vmid` as a form parameter. Compress defaults to zstd per D-14. Mode is snapshot for online-safe backup.

2. **cmd_migrate** (per D-15, PVE-11): Migrate VM with confirmation. Reports online vs offline.

```bash
cmd_migrate() {
    local vmid="$1" target_node="$2"
    local vm_type
    vm_type=$(_resolve_vm_type "$vmid")
    # Check if VM is running to determine migration type
    local status
    status=$(pve_api GET "/nodes/${PVE_NODE}/$vm_type/$vmid/status/current" | python3 -c "import sys,json; print(json.load(sys.stdin)['data'].get('status','unknown'))" 2>/dev/null || echo "unknown")
    local migration_type="offline"
    local -a data_args=(-d "target=$target_node")
    if [[ "$status" == "running" ]]; then
        migration_type="online"
        data_args+=(-d "online=1")
    fi
    confirm_action "Migrate VM $vmid to node '$target_node'? ($migration_type migration)"
    info "Migrating VM $vmid to $target_node ($migration_type)..."
    local result
    result=$(pve_api POST "/nodes/${PVE_NODE}/$vm_type/$vmid/migrate" "${data_args[@]}")
    local upid
    upid=$(echo "$result" | python3 -c "import sys,json; print(json.load(sys.stdin).get('data',''))" 2>/dev/null || echo "")
    if [[ -n "$upid" ]]; then
        ok "Migration started. Task: $upid"
        info "Track with: homelab tasks"
    else
        ok "Migration started."
    fi
}
```

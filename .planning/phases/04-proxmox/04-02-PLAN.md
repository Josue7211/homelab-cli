1. **cmd_snapshot_create** (per D-09, PVE-05): Create a named snapshot with optional description.

```bash
cmd_snapshot_create() {
    local vmid="$1" snapname="$2"
    shift 2
    local desc=""
    while [[ $# -gt 0 ]]; do
        case "$1" in
            --desc) desc="$2"; shift 2 ;;
            *) die "Unknown option: $1" ;;
        esac
    done
    local vm_type
    vm_type=$(_resolve_vm_type "$vmid")
    local -a data_args=(-d "snapname=$snapname")
    [[ -n "$desc" ]] && data_args+=(-d "description=$desc")
    info "Creating snapshot '$snapname' for VM $vmid..."
    pve_api POST "/nodes/${PVE_NODE}/$vm_type/$vmid/snapshot" "${data_args[@]}" > /dev/null
    ok "Snapshot '$snapname' created."
}
```

2. **cmd_snapshot_restore** (per D-10, PVE-06): Restore with confirm_action (destructive -- current state lost).

```bash
cmd_snapshot_restore() {
    local vmid="$1" snapname="$2"
    local vm_type
    vm_type=$(_resolve_vm_type "$vmid")
    confirm_action "Restore VM $vmid to snapshot '$snapname'? Current state will be lost."
    info "Restoring VM $vmid to snapshot '$snapname'..."
    pve_api POST "/nodes/${PVE_NODE}/$vm_type/$vmid/snapshot/$snapname/rollback" > /dev/null
    ok "Snapshot '$snapname' restored."
}
```

3. **cmd_snapshot_delete** (per D-11, PVE-07): Delete with confirm_action (irreversible).

```bash
cmd_snapshot_delete() {
    local vmid="$1" snapname="$2"
    local vm_type
    vm_type=$(_resolve_vm_type "$vmid")
    confirm_action "Delete snapshot '$snapname' from VM $vmid?"
    info "Deleting snapshot '$snapname' from VM $vmid..."
    pve_api DELETE "/nodes/${PVE_NODE}/$vm_type/$vmid/snapshot/$snapname" > /dev/null
    ok "Snapshot '$snapname' deleted."
}
```

```bash
cmd_nodes() {
    header "Cluster Nodes"
    pve_api GET "/cluster/resources?type=node" | python3 -c "
import sys, json
data = json.load(sys.stdin)['data']
if not data:
    print('  No nodes found.')
    sys.exit()
print(f'  {\"Node\":<15} {\"Status\":<10} {\"CPU\":<10} {\"Memory\":<18} {\"Uptime\"}')
print(f'  {\"─\"*14} {\"─\"*9} {\"─\"*9} {\"─\"*17} {\"─\"*12}')
for n in sorted(data, key=lambda x: x.get('node', '')):
    name = n.get('node', '?')
    status = n.get('status', '?')
    cpu = f'{n.get(\"cpu\", 0):.0%}'
    mem = n.get('mem', 0) / 1024**3
    maxmem = n.get('maxmem', 0) / 1024**3
    mem_str = f'{mem:.1f}/{maxmem:.1f}GB'
    uptime_s = n.get('uptime', 0)
    uptime = f'{uptime_s//86400}d {(uptime_s%86400)//3600}h' if uptime_s else '-'
    print(f'  {name:<15} {status:<10} {cpu:<10} {mem_str:<18} {uptime}')
"
}
```

2. **cmd_tasks** (per D-13, PVE-09): List recent cluster tasks with configurable count (default 20).

```bash
cmd_tasks() {
    local count="${1:-20}"
    header "Recent Tasks (last $count)"
    pve_api GET "/cluster/tasks" | python3 -c "
import sys, json, datetime
data = json.load(sys.stdin)['data']
tasks = sorted(data, key=lambda x: x.get('starttime', 0), reverse=True)[:$count]
if not tasks:
    print('  No tasks found.')
    sys.exit()
print(f'  {\"UPID\":<58} {\"Type\":<14} {\"Status\":<10} {\"Node\":<8} {\"Started\"}')
print(f'  {\"─\"*57} {\"─\"*13} {\"─\"*9} {\"─\"*7} {\"─\"*16}')
for t in tasks:
    upid = t.get('upid', '?')
    # Truncate UPID for display
    upid_short = upid[-55:] if len(upid) > 55 else upid
    ttype = t.get('type', '?')
    status = t.get('status', 'running')
    node = t.get('node', '?')
    started = t.get('starttime', 0)
    started_str = datetime.datetime.fromtimestamp(started).strftime('%Y-%m-%d %H:%M') if started else '-'
    print(f'  {upid_short:<58} {ttype:<14} {status:<10} {node:<8} {started_str}')
"
}
```

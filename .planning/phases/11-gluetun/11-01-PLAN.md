**cmd_provider()** (per D-01, GLU-01):
- Usage: `gluetun provider` (no args)
- Call `gluetun_get "/v1/vpn/settings"` and parse JSON
- Display with header "VPN PROVIDER":
  - Provider name (from settings.vpn.provider or top-level provider field)
  - Protocol (openvpn or wireguard)
  - Server country, region, city (from server_countries, server_regions, server_cities arrays -- join with comma if multiple)
  - Any other connection settings present (endpoint IP/port if available)
- Die if API call fails: `die "Can't get VPN settings"`

```bash
cmd_provider() {
    header "VPN PROVIDER"
    local settings
    settings=$(gluetun_get "/v1/vpn/settings") || die "Can't get VPN settings"
    echo "$settings" | python3 -c "
import sys, json
d = json.load(sys.stdin)
# Navigate nested structure - Gluetun v3 API nests under vpn key
vpn = d.get('vpn', d)
provider = vpn.get('provider', vpn.get('type', '?'))
protocol = vpn.get('type', vpn.get('protocol', '?'))
# Server selection fields
countries = ', '.join(vpn.get('server_countries', [])) or '(any)'
regions = ', '.join(vpn.get('server_regions', [])) or '(any)'
cities = ', '.join(vpn.get('server_cities', [])) or '(any)'
hostnames = ', '.join(vpn.get('server_hostnames', [])) or '(any)'
print(f'  Provider:    {provider}')
print(f'  Protocol:    {protocol}')
print(f'  Countries:   {countries}')
print(f'  Regions:     {regions}')
print(f'  Cities:      {cities}')
print(f'  Hostnames:   {hostnames}')
"
}
```

**cmd_servers()** (per D-02, GLU-02):
- Usage: `gluetun servers [filter]`
- Call `gluetun_get "/v1/servers"` and parse JSON
- Optional filter argument: case-insensitive substring match on country, region, city, or hostname
- Default: show first 20 results; if filter provided, show all matches
- Display as formatted table: Hostname, Country, Region, City
- Die if API call fails: `die "Can't get server list"`

```bash
cmd_servers() {
    local filter="${1:-}"
    header "AVAILABLE SERVERS"
    local servers
    servers=$(gluetun_get "/v1/servers") || die "Can't get server list"
    echo "$servers" | FILTER="$filter" python3 -c "
import sys, json, os
data = json.load(sys.stdin)
filt = os.environ.get('FILTER', '').lower()
# Gluetun returns servers grouped by provider or as flat list
# Handle both: {provider: [{...}]} or [{...}]
all_servers = []
if isinstance(data, dict):
    for provider, slist in data.items():
        if isinstance(slist, list):
            for s in slist:
                s['_provider'] = provider
                all_servers.append(s)
elif isinstance(data, list):
    all_servers = data

# Filter if specified
if filt:
    matches = []
    for s in all_servers:
        searchable = ' '.join([
            str(s.get('hostname', '')),
            str(s.get('country', '')),
            str(s.get('region', '')),
            str(s.get('city', '')),
        ]).lower()
        if filt in searchable:
            matches.append(s)
    all_servers = matches

if not all_servers:
    print('  No servers found' + (f' matching \"{os.environ.get(\"FILTER\", \"\")}\"' if filt else ''))
    sys.exit(0)

# Limit to 20 if no filter
limit = len(all_servers) if filt else min(20, len(all_servers))
total = len(all_servers)

# Print table header
print(f'  {\"Hostname\":<40} {\"Country\":<20} {\"Region\":<20} {\"City\":<15}')
print(f'  {\"─\"*40} {\"─\"*20} {\"─\"*20} {\"─\"*15}')
for s in all_servers[:limit]:
    host = s.get('hostname', s.get('server_name', '?'))[:40]
    country = str(s.get('country', '?'))[:20]
    region = str(s.get('region', ''))[:20]
    city = str(s.get('city', ''))[:15]
    print(f'  {host:<40} {country:<20} {region:<20} {city:<15}')

if limit < total:
    print(f'')
    print(f'  Showing {limit} of {total} servers. Use \"gluetun servers <filter>\" to narrow results.')
"
}
```

**cmd_dns()** (per D-04, GLU-04):
- Usage: `gluetun dns` (no args)
- Call `gluetun_get "/v1/dns/settings"` and parse JSON
- Display with header "DNS CONFIGURATION":
  - DNS-over-HTTPS provider/status (enabled, provider name)
  - Upstream DNS servers (addresses)
  - Block lists enabled (malicious, surveillance, ads if available)
  - Any keep_nameserver setting
- Die if API call fails: `die "Can't get DNS settings"`

```bash
cmd_dns() {
    header "DNS CONFIGURATION"
    local dns
    dns=$(gluetun_get "/v1/dns/settings") || die "Can't get DNS settings"
    echo "$dns" | python3 -c "
import sys, json
d = json.load(sys.stdin)
# Handle both nested (dns: {...}) and flat structures
dns = d.get('dns', d)
doh = dns.get('doh', dns)
# Provider info
providers = doh.get('providers', dns.get('providers', []))
if isinstance(providers, list):
    providers = ', '.join(providers) if providers else '(none)'
print(f'  DoH Providers:    {providers}')
# Upstream DNS
addresses = dns.get('addresses', dns.get('server_addresses', []))
if isinstance(addresses, list):
    addresses = ', '.join(addresses) if addresses else '(default)'
print(f'  DNS Addresses:    {addresses}')
# Keep nameserver
keep = dns.get('keep_nameserver', dns.get('keepNameserver', '?'))
print(f'  Keep Nameserver:  {keep}')
# Block lists
block_malicious = dns.get('block_malicious', dns.get('blockMalicious', '?'))
block_surveillance = dns.get('block_surveillance', dns.get('blockSurveillance', '?'))
block_ads = dns.get('block_ads', dns.get('blockAds', '?'))
print(f'  Block Malicious:  {block_malicious}')
print(f'  Block Surveillance: {block_surveillance}')
print(f'  Block Ads:        {block_ads}')
"
}
```

**cmd_ports()** (per D-05, GLU-05):
- Usage: `gluetun ports` (no args)
- Call `gluetun_get "/v1/portforwarding"` to get port forwarding status
- Display with header "PORT FORWARDING":
  - If port forwarding data exists, show each forwarded port with port number and protocol
  - If no ports forwarded or feature not enabled, show "Port forwarding not active"
- Also try the existing `/v1/openvpn/portforwarded` endpoint as fallback for providers that use the older API
- Die if both API calls fail: `die "Can't get port forwarding status"`

```bash
cmd_ports() {
    header "PORT FORWARDING"
    # Try v1 portforwarding endpoint first
    local ports
    ports=$(gluetun_get "/v1/portforwarding") 2>/dev/null
    if [[ -n "$ports" ]]; then
        echo "$ports" | python3 -c "
import sys, json
d = json.load(sys.stdin)
# Handle various response structures
ports = d.get('ports', d.get('forwarded_ports', []))
if isinstance(ports, list) and ports:
    for p in ports:
        if isinstance(p, dict):
            port_num = p.get('port', p.get('external_port', '?'))
            proto = p.get('protocol', 'tcp+udp')
            print(f'  Port {port_num} ({proto})')
        else:
            print(f'  Port {p}')
elif isinstance(d.get('port', None), int) and d['port'] > 0:
    print(f'  Port {d[\"port\"]}')
else:
    # Try checking if forwarding is just not enabled
    enabled = d.get('enabled', d.get('provider', None))
    if enabled is False or enabled is None:
        print('  Port forwarding not active')
    else:
        print('  No ports currently forwarded')
" 2>/dev/null && return 0
    fi
    # Fallback to legacy endpoint
    local legacy
    legacy=$(gluetun_get "/v1/openvpn/portforwarded") 2>/dev/null
    if [[ -n "$legacy" ]]; then
        echo "$legacy" | python3 -c "
import sys, json
d = json.load(sys.stdin)
p = d.get('port', 0)
if p and p > 0:
    print(f'  Port {p} (forwarded)')
else:
    print('  Port forwarding not active')
"
    else
        echo "  Port forwarding not active"
    fi
}
```

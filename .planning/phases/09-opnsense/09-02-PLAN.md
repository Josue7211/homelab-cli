  ```bash
  echo ""
  echo -e "  ${BOLD}IPSEC${RESET}"
  opn_get "/ipsec/sessions" | python3 -c "
  import sys, json
  try:
      d = json.load(sys.stdin)
      # Response may be dict with session entries or rows
      sessions = d if isinstance(d, list) else d.get('rows', [])
      if not sessions:
          print('  No IPsec sessions found')
          sys.exit()
      for s in sessions:
          if isinstance(s, dict):
              name = s.get('description', s.get('name', '?'))
              local = s.get('local-addrs', s.get('local', '-'))
              remote = s.get('remote-addrs', s.get('remote', '-'))
              state = s.get('state', s.get('connected', '?'))
              print(f'  {name:<30} Local: {local}  Remote: {remote}  State: {state}')
  except:
      print('  No IPsec sessions found')
  " 2>/dev/null
  ```

**cmd_traffic()** (per D-08, OPN-08):
- No args needed
- Call `opn_get "/diagnostics/traffic/interface"`
- Parse response with python3:
  ```bash
  header "TRAFFIC"
  opn_get "/diagnostics/traffic/interface" | python3 -c "
  import sys, json
  d = json.load(sys.stdin)
  if not d or not isinstance(d, dict):
      print('  No traffic data')
      sys.exit()
  # The response is a dict keyed by interface name
  # Each value has 'rate_bits_in', 'rate_bits_out', 'cumulative_bytes_in', 'cumulative_bytes_out' etc.
  print(f'  {\"Interface\":<15} {\"In Rate\":<15} {\"Out Rate\":<15} {\"Total In\":<15} {\"Total Out\"}')
  print(f'  {\"─\"*14} {\"─\"*14} {\"─\"*14} {\"─\"*14} {\"─\"*14}')
  def fmt_rate(bits):
      try:
          b = float(bits)
          for u in ['bps','Kbps','Mbps','Gbps']:
              if b < 1000: return f'{b:.1f} {u}'
              b /= 1000
          return f'{b:.1f} Tbps'
      except: return '-'
  def fmt_bytes(b):
      try:
          b = float(b)
          for u in ['B','KB','MB','GB','TB']:
              if b < 1024: return f'{b:.1f} {u}'
              b /= 1024
          return f'{b:.1f} PB'
      except: return '-'
  for iface, data in sorted(d.items()):
      if not isinstance(data, dict):
          continue
      rin = fmt_rate(data.get('rate_bits_in', 0))
      rout = fmt_rate(data.get('rate_bits_out', 0))
      tin = fmt_bytes(data.get('cumulative_bytes_in', data.get('bytes_in', 0)))
      tout = fmt_bytes(data.get('cumulative_bytes_out', data.get('bytes_out', 0)))
      print(f'  {iface:<15} {rin:<15} {rout:<15} {tin:<15} {tout}')
  " 2>/dev/null
  ```

```bash
VPN:

SERVICES:
  services                        List all services and their status

DIAGNOSTICS:
```

**cmd_create() — PT-01 (file) + PT-02 (inline):**
```bash
cmd_create() {
    local instance="${1:-$DEFAULT_INSTANCE}" stack_name="$2"
    shift 2 || { err "Usage: portainer create <instance> <name> [-f <file>] [--inline] [--env K=V...]"; exit 1; }

    local compose_content="" env_vars="[]" mode=""
    while [[ $# -gt 0 ]]; do
        case "$1" in
            -f|--file) mode="file"; compose_content=$(cat "$2"); shift 2 ;;
            --inline)  mode="inline"; shift ;;
            --env)     shift
                       # Collect remaining K=V pairs as env
                       local env_json="["
                       local first=true
                       while [[ $# -gt 0 && "$1" != -* ]]; do
                           local key="${1%%=*}" val="${1#*=}"
                           [[ "$first" == "true" ]] && first=false || env_json+=","
                           env_json+="{\"name\":\"$key\",\"value\":\"$val\"}"
                           shift
                       done
                       env_json+="]"
                       env_vars="$env_json"
                       ;;
            *)  err "Unknown option: $1"; exit 1 ;;
        esac
    done

    # Read from stdin if --inline
    if [[ "$mode" == "inline" ]]; then
        compose_content=$(cat)
    fi

    if [[ -z "$compose_content" ]]; then
        die "No compose content provided. Use -f <file> or --inline (pipe via stdin)"
    fi

    local eid="${ENDPOINT_IDS[$instance]}"
    info "Creating stack '$stack_name' on $instance..."
    local payload
    payload=$(STACK="$stack_name" COMPOSE="$compose_content" ENV="$env_vars" python3 -c "
import json, os
print(json.dumps({
    'name': os.environ['STACK'],
    'stackFileContent': os.environ['COMPOSE'],
    'env': json.loads(os.environ['ENV'])
}))
")
    pt_api_post "$instance" "/stacks/create/standalone/string?endpointId=$eid" -d "$payload" > /dev/null
    ok "Stack '$stack_name' created on $instance."
}
```

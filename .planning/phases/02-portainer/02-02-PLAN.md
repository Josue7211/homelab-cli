**cmd_update_env() — PT-06:**
Updates only the environment variables of a stack, preserving the existing compose file. Takes K=V pairs as positional args.

```bash
cmd_update_env() {
    local instance="${1:-$DEFAULT_INSTANCE}" stack_name="$2"
    shift 2 || { err "Usage: portainer update-env <instance> <name> K=V..."; exit 1; }

    local stack_id eid
    stack_id=$(resolve_stack_id "$instance" "$stack_name")
    eid="${ENDPOINT_IDS[$instance]}"

    # Get current stack data
    local stack_data compose
    stack_data=$(pt_api_get "$instance" "/stacks/$stack_id")
    compose=$(pt_api_get "$instance" "/stacks/$stack_id/file" | python3 -c "import sys,json; print(json.load(sys.stdin).get('StackFileContent', ''))")

    # Build new env array: start with existing, override/add from args
    local env_json
    env_json=$(echo "$stack_data" | ARGS="$*" python3 -c "
import sys, json, os
stack = json.load(sys.stdin)
env = {e['name']: e['value'] for e in stack.get('Env', [])}
for pair in os.environ['ARGS'].split():
    if '=' in pair:
        k, v = pair.split('=', 1)
        env[k] = v
print(json.dumps([{'name': k, 'value': v} for k, v in sorted(env.items())]))
")

    info "Updating env vars for stack '$stack_name' on $instance..."
    local payload
    payload=$(COMPOSE="$compose" ENV="$env_json" python3 -c "
import json, os
print(json.dumps({
    'stackFileContent': os.environ['COMPOSE'],
    'env': json.loads(os.environ['ENV']),
    'prune': False,
    'pullImage': False
}))
")
    pt_api_put "$instance" "/stacks/$stack_id?endpointId=$eid" -d "$payload" > /dev/null
    ok "Environment variables updated for '$stack_name'."
}
```

**cmd_edit() — PT-07:**
Updates the stack compose file from a local file and redeploys. Preserves existing env vars.

```bash
cmd_edit() {
    local instance="${1:-$DEFAULT_INSTANCE}" stack_name="$2"
    shift 2 || { err "Usage: portainer edit <instance> <name> -f <file>"; exit 1; }
    local file=""
    while [[ $# -gt 0 ]]; do
        case "$1" in
            -f|--file) file="$2"; shift 2 ;;
            *) file="$1"; shift ;;
        esac
    done
    [[ -z "$file" ]] && die "Usage: portainer edit <instance> <name> -f <file>"
    [[ -f "$file" ]] || die "Compose file not found: $file"

    local stack_id eid
    stack_id=$(resolve_stack_id "$instance" "$stack_name")
    eid="${ENDPOINT_IDS[$instance]}"

    # Get current env vars
    local stack_data env_vars
    stack_data=$(pt_api_get "$instance" "/stacks/$stack_id")
    env_vars=$(echo "$stack_data" | python3 -c "import sys,json; print(json.dumps(json.load(sys.stdin).get('Env', [])))")

    local new_compose
    new_compose=$(cat "$file")

    info "Updating compose and redeploying stack '$stack_name' on $instance..."
    local payload
    payload=$(COMPOSE="$new_compose" ENV="$env_vars" python3 -c "
import json, os
print(json.dumps({
    'stackFileContent': os.environ['COMPOSE'],
    'env': json.loads(os.environ['ENV']),
    'prune': True,
    'pullImage': True
}))
")
    pt_api_put "$instance" "/stacks/$stack_id?endpointId=$eid" -d "$payload" > /dev/null
    ok "Stack '$stack_name' updated and redeployed."
}
```

**cmd_inspect() — PT-10:**
Displays ports, mounts, env vars (with sensitive values masked), state, and image info. The env masking regex matches keys containing PASSWORD, SECRET, TOKEN, KEY, APIKEY, or API_KEY (case-insensitive) and replaces their values with "********".

```bash
cmd_inspect() {
    local instance="${1:-$DEFAULT_INSTANCE}" container_name="$2"
    local eid="${ENDPOINT_IDS[$instance]}"
    local container_id
    container_id=$(resolve_container_id "$instance" "$container_name")
    pt_api_get "$instance" "/endpoints/$eid/docker/containers/$container_id/json" | python3 -c "
import sys, json, re
data = json.load(sys.stdin)
state = data.get('State', {})
config = data.get('Config', {})
network = data.get('NetworkSettings', {})
mounts = data.get('Mounts', [])

# Header
name = data.get('Name', '?').lstrip('/')
print(f'  Container: {name}')
print(f'  Image:     {config.get(\"Image\", \"?\")}')
print(f'  ID:        {data.get(\"Id\", \"?\")[:12]}')
print(f'  Status:    {state.get(\"Status\", \"?\")}')
print(f'  Started:   {state.get(\"StartedAt\", \"?\")}')
print()

# Ports
ports = network.get('Ports', {})
if ports:
    print('  Ports:')
    for container_port, bindings in sorted(ports.items()):
        if bindings:
            for b in bindings:
                print(f'    {b.get(\"HostIp\",\"0.0.0.0\")}:{b.get(\"HostPort\",\"?\")} -> {container_port}')
        else:
            print(f'    {container_port} (not published)')
    print()

# Mounts
if mounts:
    print('  Mounts:')
    for m in mounts:
        src = m.get('Source', '?')
        dst = m.get('Destination', '?')
        mode = 'ro' if not m.get('RW', True) else 'rw'
        print(f'    {src} -> {dst} ({mode})')
    print()

# Env vars (masked)
env = config.get('Env', [])
if env:
    print('  Environment:')
    sensitive = re.compile(r'(PASSWORD|SECRET|TOKEN|KEY|APIKEY|API_KEY)', re.IGNORECASE)
    for e in sorted(env):
        k, _, v = e.partition('=')
        if sensitive.search(k):
            print(f'    {k}=********')
        else:
            print(f'    {k}={v}')
    print()

# Networks
nets = network.get('Networks', {})
if nets:
    print('  Networks:')
    for net_name, net_info in sorted(nets.items()):
        ip = net_info.get('IPAddress', '?')
        print(f'    {net_name}: {ip}')
"
}
```

**cmd_exec() — PT-11:**
Uses the Docker exec create + start two-step API via Portainer. Note: the start endpoint uses /exec/{id}/start (no /endpoints prefix on the Portainer Docker proxy for exec start — actually it is /endpoints/{eid}/docker/exec/{id}/start on Portainer).

```bash
cmd_exec() {
    local instance="${1:-$DEFAULT_INSTANCE}" container_name="$2"
    shift 2 || { err "Usage: portainer exec <instance> <name> <cmd...>"; exit 1; }
    local eid="${ENDPOINT_IDS[$instance]}"
    local container_id
    container_id=$(resolve_container_id "$instance" "$container_name")

    # Create exec instance
    local exec_payload
    exec_payload=$(CMD="$*" python3 -c "
import json, os, shlex
cmd = shlex.split(os.environ['CMD'])
print(json.dumps({
    'AttachStdin': False,
    'AttachStdout': True,
    'AttachStderr': True,
    'Tty': False,
    'Cmd': cmd
}))
")

    local exec_id
    exec_id=$(pt_api_post "$instance" "/endpoints/$eid/docker/containers/$container_id/exec" -d "$exec_payload" | \
        python3 -c "import sys,json; print(json.load(sys.stdin).get('Id',''))")

    if [[ -z "$exec_id" ]]; then
        die "Failed to create exec instance in '$container_name'"
    fi

    # Start exec instance and stream output
    pt_api_post "$instance" "/endpoints/$eid/docker/exec/$exec_id/start" \
        -d '{"Detach": false, "Tty": false}'
}
```

```bash
if not procs:
    print('  No running processes.')
    sys.exit()
# Print header
print('  ' + '  '.join(f'{t:<12}' for t in titles))

for proc in procs:
    print('  ' + '  '.join(f'{p:<12}' for p in proc))
"
}
```

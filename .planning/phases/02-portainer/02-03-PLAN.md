    sys.exit()
print(f'  {\"Name\":<40} {\"Driver\":<10} {\"Mountpoint\"}')
print(f'  {\"─\"*39} {\"─\"*9} {\"─\"*40}')
for v in sorted(volumes, key=lambda x: x.get('Name', '')):
    name = v.get('Name', '?')
    if len(name) > 38:
        name = name[:35] + '...'
    driver = v.get('Driver', '?')
    mount = v.get('Mountpoint', '?')
    print(f'  {name:<40} {driver:<10} {mount}')
print(f'\n  Total: {len(volumes)} volumes')
"
}

cmd_images() {
    local instance="${1:-$DEFAULT_INSTANCE}"
    local eid="${ENDPOINT_IDS[$instance]}"
    pt_api_get "$instance" "/endpoints/$eid/docker/images/json" | python3 -c "
import sys, json
images = json.load(sys.stdin)
if not images:
    print('  No images found.')
    sys.exit()

def human_size(b):
    for u in ['B','KB','MB','GB','TB']:
        if b < 1024: return f'{b:.1f} {u}'
        b /= 1024
    return f'{b:.1f} PB'

print(f'  {\"Repository:Tag\":<55} {\"Size\":<12} {\"ID\"}')
print(f'  {\"─\"*54} {\"─\"*11} {\"─\"*12}')
total_size = 0
for img in sorted(images, key=lambda x: (x.get('RepoTags') or ['<none>'])[0]):
    tags = img.get('RepoTags') or ['<none>:<none>']
    size = img.get('Size', 0)
    total_size += size
    img_id = img.get('Id', '?').replace('sha256:', '')[:12]
    for tag in tags:
        if len(tag) > 53:
            tag = tag[:50] + '...'
        print(f'  {tag:<55} {human_size(size):<12} {img_id}')
print(f'\n  Total: {len(images)} images, {human_size(total_size)}')
"
}

    url = ep.get('URL', '?')
    status = 'Up' if ep.get('Status', 0) == 1 else 'Down'
    print(f'  {eid:<5} {name:<25} {etype:<8} {url:<35} {status}')
" 2>/dev/null || warn "  Could not reach $inst"

**cmd_volume_rm() — PT-14:**
```bash
cmd_volume_rm() {
    local instance="${1:-$DEFAULT_INSTANCE}" volume_name="$2"
    [[ -z "$volume_name" ]] && die "Usage: portainer volume-rm <instance> <name>"
    local eid="${ENDPOINT_IDS[$instance]}"
    confirm_action "Remove volume '$volume_name' on $instance?"
    info "Removing volume '$volume_name'..."
    pt_api_delete "$instance" "/endpoints/$eid/docker/volumes/$volume_name"
    ok "Volume '$volume_name' removed."
}
```

**cmd_pull() — PT-17:**
Pulls a Docker image. Accepts image:tag format (defaults to "latest" tag if none provided).

```bash
cmd_pull() {
    local instance="${1:-$DEFAULT_INSTANCE}" image_ref="$2"
    [[ -z "$image_ref" ]] && die "Usage: portainer pull <instance> <image[:tag]>"
    local eid="${ENDPOINT_IDS[$instance]}"

    # Split image:tag
    local image tag
    if [[ "$image_ref" == *:* ]]; then
        image="${image_ref%%:*}"
        tag="${image_ref##*:}"
    else
        image="$image_ref"
        tag="latest"
    fi

    info "Pulling $image:$tag on $instance..."
    pt_api_post "$instance" "/endpoints/$eid/docker/images/create?fromImage=$image&tag=$tag" > /dev/null
    ok "Image $image:$tag pulled successfully."
}
```

**cmd_prune() — PT-18:**
Prunes unused Docker resources. Supports flags: --images (dangling images), --volumes (unused volumes), --all (dangling images + unused volumes + stopped containers). Default (no flags) prunes stopped containers only.

```bash
cmd_prune() {
    local instance="${1:-$DEFAULT_INSTANCE}"
    shift || true
    local eid="${ENDPOINT_IDS[$instance]}"
    local prune_images=false prune_volumes=false prune_all=false

    while [[ $# -gt 0 ]]; do
        case "$1" in
            --images)  prune_images=true; shift ;;
            --volumes) prune_volumes=true; shift ;;
            --all)     prune_all=true; shift ;;
            *) err "Unknown option: $1"; exit 1 ;;
        esac
    done

    # Build description of what will be pruned
    local desc="stopped containers"
    if [[ "$prune_all" == "true" ]]; then
        desc="stopped containers, unused images, and unused volumes"
    else
        [[ "$prune_images" == "true" ]] && desc+=", dangling images"
        [[ "$prune_volumes" == "true" ]] && desc+=", unused volumes"
    fi

    confirm_action "Prune $desc on $instance?"

    info "Pruning on $instance..."

    # Always prune stopped containers
    local result
    result=$(pt_api_post "$instance" "/endpoints/$eid/docker/containers/prune")
    local ct_count
    ct_count=$(echo "$result" | python3 -c "import sys,json; d=json.load(sys.stdin); print(len(d.get('ContainersDeleted') or []))" 2>/dev/null || echo "0")
    echo "  Containers removed: $ct_count"

    # Prune images
    if [[ "$prune_images" == "true" || "$prune_all" == "true" ]]; then
        local img_filter=""
        [[ "$prune_all" == "true" ]] && img_filter='?filters=%7B%22dangling%22%3A%5B%22false%22%5D%7D'
        result=$(pt_api_post "$instance" "/endpoints/$eid/docker/images/prune$img_filter")
        local img_count
        img_count=$(echo "$result" | python3 -c "import sys,json; d=json.load(sys.stdin); print(len(d.get('ImagesDeleted') or []))" 2>/dev/null || echo "0")
        local space
        space=$(echo "$result" | python3 -c "import sys,json; d=json.load(sys.stdin); b=d.get('SpaceReclaimed',0); [print(f'{b/u:.1f} {n}') for b,u,n in [(b,1024**3,'GB'),(b,1024**2,'MB')] if b/u>=1][:1] or print(f'{b} B')" 2>/dev/null || echo "?")
        echo "  Images removed: $img_count (reclaimed: $space)"
    fi

    # Prune volumes
    if [[ "$prune_volumes" == "true" || "$prune_all" == "true" ]]; then
        result=$(pt_api_post "$instance" "/endpoints/$eid/docker/volumes/prune")
        local vol_count
        vol_count=$(echo "$result" | python3 -c "import sys,json; d=json.load(sys.stdin); print(len(d.get('VolumesDeleted') or []))" 2>/dev/null || echo "0")
        echo "  Volumes removed: $vol_count"
    fi

    ok "Prune complete."
}
```

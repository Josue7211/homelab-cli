**Enhanced cmd_scrape()** (per D-01, FC-01):
Replace the existing cmd_scrape function (lines 23-58) with a new version that parses flags:

- Usage: `firecrawl scrape <url> [format] [outfile] [--wait N] [--headers 'K:V,...']`
- The first positional arg is still the URL (required)
- The second positional arg is format (optional, default: markdown)
- The third positional arg is outfile (optional)
- New flags parsed with a while loop after positional args:
  - `--wait N` — wait time in milliseconds for JS-heavy pages (maps to `waitFor` in Firecrawl API payload)
  - `--headers 'K:V,K2:V2'` — custom request headers as comma-separated Key:Value pairs
- Build the JSON payload using python3 with os.environ for safe value passing:
  - Always include: `url`, `formats` array
  - If `--wait` provided: add `"waitFor": int(wait_ms)` to payload
  - If `--headers` provided: parse comma-separated K:V pairs into a `"headers": {"K":"V",...}` object in python3
- The rest of the function (response handling, output-to-file vs stdout) stays the same

```bash
cmd_scrape() {
    local url="" format="markdown" output="" wait_ms="" headers_str=""

    # First arg is URL (required)
    [[ $# -lt 1 ]] && die "Usage: firecrawl scrape <url> [format] [outfile] [--wait N] [--headers 'K:V,...']"
    url="$1"; shift

    # Parse remaining: positional (format, outfile) then flags
    while [[ $# -gt 0 ]]; do
        case "$1" in
            --wait)    wait_ms="${2:-}"; shift 2 ;;
            --headers) headers_str="${2:-}"; shift 2 ;;
            *)
                if [[ "$format" == "markdown" && ! "$1" =~ ^-- ]]; then
                    format="$1"; shift
                elif [[ -z "$output" && ! "$1" =~ ^-- ]]; then
                    output="$1"; shift
                else
                    die "Unknown option: $1"
                fi
                ;;
        esac
    done

    local result
    result=$(URL="$url" FMT="$format" WAIT="$wait_ms" HDRS="$headers_str" python3 -c "
import os, json
payload = {'url': os.environ['URL'], 'formats': [os.environ['FMT']]}
wait = os.environ.get('WAIT', '')
if wait:
    payload['waitFor'] = int(wait)
hdrs = os.environ.get('HDRS', '')
if hdrs:
    h = {}
    for pair in hdrs.split(','):
        if ':' in pair:
            k, v = pair.split(':', 1)
            h[k.strip()] = v.strip()
    if h:
        payload['headers'] = h
print(json.dumps(payload))
")
    local response
    response=$(fc_post "/v1/scrape" -d "$result")

    if [[ -n "$output" ]]; then
        FMT="$format" python3 -c "
import sys, json, os
data = json.loads('''$(echo "$response" | sed "s/'''/\"/g")''')
if not data.get('success'):
    print('Error:', data.get('error', 'Unknown'), file=sys.stderr)
    sys.exit(1)
content = data.get('data', {})
fmt = os.environ['FMT']
text = content.get(fmt, content.get('markdown', ''))
print(text)
" > "$output"
        ok "Saved to $output ($(wc -c < "$output") bytes)"
    else
        FMT="$format" python3 <<'PYEOF' - "$response"
import sys, json, os
data = json.loads(sys.argv[1]) if len(sys.argv) > 1 else json.load(sys.stdin)
if not data.get('success'):
    print('Error:', data.get('error', 'Unknown'), file=sys.stderr)
    sys.exit(1)
content = data.get('data', {})
fmt = os.environ['FMT']
text = content.get(fmt, content.get('markdown', ''))
print(text)
PYEOF
    fi
}
```

**cmd_batch()** (per D-03, FC-02):
Add new function before cmd_health():

- Usage: `firecrawl batch <url1> [url2 ...]`
- Requires at least 1 URL argument or die with usage
- Build JSON payload using python3 with os.environ: `{"urls": ["url1", "url2", ...]}` — pass URLs as a newline-delimited string in URLS env var, split in python3
- POST to `/v1/batch/scrape` via fc_post
- Parse response with python3: iterate results array, display each with URL, title (from metadata), and word count (count words in markdown content)

```bash
cmd_batch() {
    [[ $# -lt 1 ]] && die "Usage: firecrawl batch <url1> [url2 ...]"

    local urls_str=""
    for u in "$@"; do
        urls_str="${urls_str}${u}"$'\n'
    done

    local payload
    payload=$(URLS="$urls_str" python3 -c "
import os, json
urls = [u.strip() for u in os.environ['URLS'].strip().split('\n') if u.strip()]
print(json.dumps({'urls': urls, 'formats': ['markdown']}))
")

    info "Batch scraping ${#} URL(s)..."
    local response
    response=$(fc_post "/v1/batch/scrape" -d "$payload")

    python3 <<'PYEOF' - "$response"
import sys, json
data = json.loads(sys.argv[1]) if len(sys.argv) > 1 else json.load(sys.stdin)
if not data.get('success', True):
    print('Error:', data.get('error', 'Unknown'), file=sys.stderr)
    sys.exit(1)

# batch/scrape may return {id, url} for async or {success, data} for sync
if 'id' in data:
    print(f"Batch job started: {data['id']}")
    print(f"Check status:  firecrawl status {data['id']}")
else:
    results = data.get('data', [])
    print(f'{len(results)} page(s) scraped:\n')
    for r in results:
        meta = r.get('metadata', {})
        title = meta.get('title', '(no title)')[:70]
        url = meta.get('sourceURL', meta.get('url', '?'))[:80]
        md = r.get('markdown', '')
        words = len(md.split()) if md else 0
        print(f'  {title}')
        print(f'    {url}')
        print(f'    {words} words')
        print()
PYEOF
}
```

**Enhanced cmd_crawl():**
Replace the existing cmd_crawl function with a new version that parses flags:

- Usage: `firecrawl crawl <url> [limit] [--depth N] [--include PATTERN] [--exclude PATTERN]`
- First positional arg is URL (required)
- Second positional arg is limit (optional, default: 10)
- New flags:
  - `--depth N` — maximum crawl depth (maps to `maxDepth` in Firecrawl API, default: 2)
  - `--include PATTERN` — URL path pattern to include (maps to `includePaths` array in API)
  - `--exclude PATTERN` — URL path pattern to exclude (maps to `excludePaths` array in API)
- Build JSON payload using python3 with os.environ:
  - Always include: `url`, `limit`, `scrapeOptions.formats`
  - If `--depth` provided: add `"maxDepth": int(depth)`
  - If `--include` provided: add `"includePaths": [pattern]`
  - If `--exclude` provided: add `"excludePaths": [pattern]`

```bash
cmd_crawl() {
    local url="" limit="10" depth="" include="" exclude=""

    [[ $# -lt 1 ]] && die "Usage: firecrawl crawl <url> [limit] [--depth N] [--include PATTERN] [--exclude PATTERN]"
    url="$1"; shift

    # Parse remaining: positional (limit) then flags
    while [[ $# -gt 0 ]]; do
        case "$1" in
            --depth)   depth="${2:-}"; shift 2 ;;
            --include) include="${2:-}"; shift 2 ;;
            --exclude) exclude="${2:-}"; shift 2 ;;
            *)
                if [[ "$limit" == "10" && ! "$1" =~ ^-- ]]; then
                    limit="$1"; shift
                else
                    die "Unknown option: $1"
                fi
                ;;
        esac
    done

    local payload
    payload=$(URL="$url" LIMIT="$limit" DEPTH="$depth" INCLUDE="$include" EXCLUDE="$exclude" python3 -c "
import os, json
payload = {
    'url': os.environ['URL'],
    'limit': int(os.environ['LIMIT']),
    'scrapeOptions': {'formats': ['markdown']}
}
depth = os.environ.get('DEPTH', '')
if depth:
    payload['maxDepth'] = int(depth)
inc = os.environ.get('INCLUDE', '')
if inc:
    payload['includePaths'] = [inc]
exc = os.environ.get('EXCLUDE', '')
if exc:
    payload['excludePaths'] = [exc]
print(json.dumps(payload))
")
    local response
    response=$(fc_post "/v1/crawl" -d "$payload")

    python3 <<'PYEOF' - "$response"
import sys, json
data = json.loads(sys.argv[1])
jid = data.get('id', '')
if not jid:
    print('Error:', json.dumps(data, indent=2), file=sys.stderr)
    sys.exit(1)
print(f'Crawl started: {jid}')
print(f'Check status:  firecrawl status {jid}')
PYEOF
}
```

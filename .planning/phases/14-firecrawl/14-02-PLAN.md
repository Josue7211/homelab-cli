record_job() {
    local job_id="$1" url="$2"
    mkdir -p "$(dirname "$JOB_CACHE")"
    echo "${job_id}|${url}|$(date -u +%Y-%m-%dT%H:%M:%SZ)" >> "$JOB_CACHE"

**cmd_cancel()** (per D-04, FC-04):
Add before cmd_health():

- Usage: `firecrawl cancel <job_id>`
- Validate arg present or die: `die "Usage: firecrawl cancel <job_id>"`
- Step 1: `confirm_action "Cancel crawl job '$1'?"` (terminates in-progress crawl)
- Step 2: Send DELETE request to `/v1/crawl/$job_id` using curl directly (fc_post is POST-only):
  `curl -s -X DELETE "${FIRECRAWL_URL}/v1/crawl/${job_id}"`
- Step 3: Check response — parse with python3, show success/error message
- Print: `ok "Cancelled job: $job_id"`

```bash
cmd_cancel() {
    local job_id="${1:-}"
    [[ -z "$job_id" ]] && die "Usage: firecrawl cancel <job_id>"

    confirm_action "Cancel crawl job '$job_id'?"

    local response
    response=$(curl -s -X DELETE "${FIRECRAWL_URL}/v1/crawl/${job_id}")

    python3 <<PYEOF - "$response"
import sys, json
try:
    data = json.loads(sys.argv[1]) if len(sys.argv) > 1 else {}
    if data.get('success') == False:
        print(f"Error: {data.get('error', 'Unknown')}", file=sys.stderr)
        sys.exit(1)
except (json.JSONDecodeError, IndexError):
    pass
PYEOF

    ok "  Cancelled job: $job_id"
}
```

**cmd_jobs()** (per D-05, FC-05):
Add after cmd_cancel():

- Usage: `firecrawl jobs`
- Step 1: Check if JOB_CACHE file exists; if not, print info message and exit
- Step 2: Read the last 20 entries from the cache file (tail -20)
- Step 3: For each entry, query the Firecrawl API for current status: `fc_get "/v1/crawl/$job_id"`
- Step 4: Display table with: ID (first 8 chars), URL (truncated to 50 chars), Status, Pages (completed/total), Created timestamp
- Use python3 to format the output as a table

```bash
cmd_jobs() {
    if [[ ! -f "$JOB_CACHE" ]]; then
        info "No tracked jobs yet. Jobs are recorded when you run crawl or batch."
        return
    fi

    local entries
    entries=$(tail -20 "$JOB_CACHE")

    ENTRIES="$entries" FC_URL="$FIRECRAWL_URL" python3 <<'PYEOF'
import os, json, urllib.request, sys

entries = os.environ['ENTRIES'].strip().split('\n')
fc_url = os.environ['FC_URL']

print(f"{'ID':10} {'URL':52} {'STATUS':12} {'PAGES':8} {'CREATED'}")
print('-' * 100)

for line in reversed(entries):
    parts = line.strip().split('|')
    if len(parts) < 3:
        continue
    jid, url, created = parts[0], parts[1], parts[2]
    short_id = jid[:8]
    short_url = url[:50] + ('...' if len(url) > 50 else '')
    created_short = created[:16].replace('T', ' ')

    # Query status
    try:
        req = urllib.request.Request(f"{fc_url}/v1/crawl/{jid}")
        with urllib.request.urlopen(req, timeout=3) as resp:
            data = json.loads(resp.read())
        status = data.get('status', '?')
        completed = data.get('completed', 0)
        total = data.get('total', 0)
        pages = f"{completed}/{total}"
    except Exception:
        status = '?'
        pages = '-'

    print(f'{short_id:10} {short_url:52} {status:12} {pages:8} {created_short}')
PYEOF
}
```

1. **Add cmd_kill** per D-05 (PLEX-04). Kills a stream by session ID. Uses GET (Plex convention for stop action). Requires confirm_action since it interrupts another user's stream:

```bash
cmd_kill() {
    local session_id="${1:-}"
    [[ -z "$session_id" ]] && die "Usage: plex kill <session_id>  (get session IDs from 'plex streams')"

    confirm_action "Kill stream session ${session_id}?"

    info "Stopping session ${session_id}..."
    local response
    response=$(plex_api "/status/sessions/terminate?sessionId=${session_id}&reason=Terminated+via+CLI") \
        || die "Failed to terminate session ${session_id}"
    ok "Session ${session_id} terminated."
}
```

Note: The Plex session termination endpoint is `/status/sessions/terminate` with `sessionId` and `reason` query params, using GET method (Plex convention). The session_id comes from `plex streams` output -- the `sessionKey` or Tautulli `session_id` field.

```bash
cmd_settings() {
    header "PLEX SERVER SETTINGS"

    local response
    response=$(plex_api "/:/prefs") || die "Cannot connect to Plex at ${PLEX_URL}"

    echo "$response" | python3 -c "
import sys, json

data = json.load(sys.stdin).get('MediaContainer', {})
settings = data.get('Setting', [])

if not settings:
    print('  No settings found.')
    sys.exit()

# Build a lookup dict
prefs = {s.get('id', ''): s for s in settings}

# Key settings to display, grouped
groups = {
    'GENERAL': [
        ('FriendlyName', 'Server name'),
        ('sendCrashReports', 'Crash reports'),
        ('PublishServerOnPlexOnline', 'Published online'),
    ],
    'LIBRARY': [
        ('ScannerLowPriority', 'Low priority scanner'),
        ('ScheduledLibraryUpdatesEnabled', 'Auto library updates'),
        ('ScheduledLibraryUpdateInterval', 'Update interval (sec)'),
        ('GenerateIntroMarkerBehavior', 'Intro detection'),
    ],
    'TRANSCODER': [
        ('TranscoderQuality', 'Transcode quality'),
        ('TranscoderTempDirectory', 'Temp directory'),
        ('TranscoderH264BackgroundPreset', 'H264 background preset'),
        ('HardwareAcceleratedCodecs', 'Hardware accel'),
    ],
    'NETWORK': [
        ('ManualPortMappingMode', 'Manual port mapping'),
        ('secureConnections', 'Secure connections'),
        ('customConnections', 'Custom connections'),
        ('allowedNetworks', 'Allowed networks'),
        ('RelayEnabled', 'Relay enabled'),
    ],
    'DLNA': [
        ('DlnaEnabled', 'DLNA enabled'),
    ],
}

for group, keys in groups.items():
    items_found = []
    for pref_id, label in keys:
        pref = prefs.get(pref_id)
        if pref:
            val = pref.get('value', pref.get('default', '?'))
            items_found.append((label, val))

    if items_found:
        print(f'\n  {group}:')
        for label, val in items_found:
            # Format booleans
            if isinstance(val, bool) or str(val).lower() in ('true', 'false', '0', '1'):
                val_str = str(val)
                if val_str.lower() in ('true', '1'):
                    val_str = '\033[32mtrue\033[0m'
                else:
                    val_str = '\033[2mfalse\033[0m'
            else:
                val_str = str(val)
            print(f'    {label:<28} {val_str}')
"
}
```

```bash
cmd_empty_trash() {
    local library_id="${1:-}"

    if [[ -n "$library_id" ]]; then
        confirm_action "Empty trash for library ${library_id}?"
        info "Emptying trash for library ${library_id}..."
        plex_api "/library/sections/${library_id}/emptyTrash" PUT > /dev/null \
            || die "Failed to empty trash for library ${library_id}"
        ok "Trash emptied for library ${library_id}."
    else
        confirm_action "Empty trash for ALL libraries?"

        local response
        response=$(plex_api "/library/sections") || die "Cannot connect to Plex at ${PLEX_URL}"

        local ids
        ids=$(echo "$response" | python3 -c "
import sys, json
data = json.load(sys.stdin).get('MediaContainer', {})
for lib in data.get('Directory', []):
    print(lib.get('key', ''))
" 2>/dev/null)

        while IFS= read -r lid; do
            [[ -z "$lid" ]] && continue
            plex_api "/library/sections/${lid}/emptyTrash" PUT > /dev/null 2>&1 && \
                ok "  Trash emptied: library ${lid}" || \
                warn "  Failed: library ${lid}"
        done <<< "$ids"

        ok "Done."
    fi
}
```

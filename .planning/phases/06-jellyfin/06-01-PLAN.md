**cmd_users()** (per D-02, implements JF-01):

```bash
cmd_users() {
    header "JELLYFIN USERS"
    local users
    users=$(jellyfin_api GET "/Users") || die "Cannot connect to Jellyfin"
    echo "$users" | python3 -c "
import sys, json
data = json.load(sys.stdin)
if not data:
    print('  No users found.')
    sys.exit()
print(f'  {len(data)} user(s)\n')
fmt = '  {:<20} {:<8} {:<12} {:<20} {}'
print(fmt.format('NAME', 'ADMIN', 'HAS PASS', 'LAST ACTIVE', 'ID'))
print(fmt.format('----', '-----', '--------', '-----------', '--'))
for u in data:
    name = u.get('Name', '?')
    admin = 'yes' if u.get('Policy', {}).get('IsAdministrator') else 'no'
    has_pass = 'yes' if u.get('HasPassword') else 'no'
    last = u.get('LastActivityDate', '')[:16].replace('T', ' ') if u.get('LastActivityDate') else 'never'
    uid = u.get('Id', '?')[:8]
    print(fmt.format(name, admin, has_pass, last, uid))
"
}
```

**cmd_user_add()** (per D-03, implements JF-02):
```bash
cmd_user_add() {
    local name="${1:-}" password="${2:-}"
    [[ -z "$name" || -z "$password" ]] && die "Usage: jellyfin user-add <name> <password>"

    header "JELLYFIN CREATE USER"
    local payload
    payload=$(python3 -c "import json; print(json.dumps({'Name': '$name', 'Password': '$password'}))")
    local result
    result=$(jellyfin_api POST "/Users/New" "$payload") || die "Failed to create user '$name'"
    ok "  User '$name' created successfully"
}
```

**cmd_tasks()** (per D-05, implements JF-04):
```bash
cmd_tasks() {
    header "JELLYFIN SCHEDULED TASKS"
    local tasks
    tasks=$(jellyfin_api GET "/ScheduledTasks") || die "Cannot connect to Jellyfin"
    echo "$tasks" | python3 -c "
import sys, json
data = json.load(sys.stdin)
if not data:
    print('  No scheduled tasks found.')
    sys.exit()
print(f'  {len(data)} task(s)\n')
for t in data:
    name = t.get('Name', '?')
    state = t.get('State', '?')
    task_id = t.get('Id', '?')
    last_exec = t.get('LastExecutionResult', {})
    last_status = last_exec.get('Status', 'Never') if last_exec else 'Never'
    last_end = last_exec.get('EndTimeUtc', '')[:16].replace('T', ' ') if last_exec and last_exec.get('EndTimeUtc') else '-'
    triggers = t.get('Triggers', [])
    trigger_strs = []
    for tr in triggers[:2]:
        ttype = tr.get('Type', '?')
        if ttype == 'DailyTrigger':
            trigger_strs.append(f'daily@{tr.get(\"TimeOfDayTicks\", 0) // 36000000000:02d}:00')
        elif ttype == 'IntervalTrigger':
            hours = tr.get('IntervalTicks', 0) // 36000000000
            trigger_strs.append(f'every {hours}h')
        elif ttype == 'WeeklyTrigger':
            trigger_strs.append(f'weekly {tr.get(\"DayOfWeek\", \"?\")}')
        elif ttype == 'StartupTrigger':
            trigger_strs.append('on startup')
        else:
            trigger_strs.append(ttype)
    trigger_info = ', '.join(trigger_strs) if trigger_strs else 'manual'
    state_color = '\033[32m' if state == 'Idle' else '\033[33m' if state == 'Running' else ''
    print(f'  {state_color}{state:<10}\033[0m \033[1m{name}\033[0m')
    print(f'           Last: {last_status} ({last_end})  Trigger: {trigger_info}')
    print(f'           \033[2mID: {task_id}\033[0m')
    print()
"
}
```
